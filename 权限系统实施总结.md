实施方案:
```python
# students.py
from utils.permission_checker import can_access_students, get_accessible_classes

@students_bp.route('/api/students', methods=['GET'])
@login_required
def get_all_students():
    # 获取用户有权限访问的班级列表
    accessible_classes = get_accessible_classes(current_user)
    
    if not accessible_classes:
        return jsonify({
            'status': 'ok',
            'count': 0,
            'students': [],
            'message': '您没有权限查看任何班级的学生'
        })
    
    # 只查询有权限的班级
    conn = get_db_connection()
    cursor = conn.cursor()
    placeholders = ','.join('?' * len(accessible_classes))
    cursor.execute(f'''
        SELECT s.*, c.class_name 
        FROM students s
        LEFT JOIN classes c ON s.class_id = c.id
        WHERE s.class_id IN ({placeholders})
        ORDER BY CAST(s.id AS INTEGER)
    ''', accessible_classes)
    # ...
```
